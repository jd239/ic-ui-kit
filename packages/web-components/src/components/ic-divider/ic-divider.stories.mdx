  "border-style": "solid",

  "label-placement": "none",

  orientation: "horizontal",
  theme: "inherit",
  weight: "thin",
  "--ic-divider-horizontal-width": "100%",

  "--ic-divider-background": "",
  "--ic-divider-label-width": "",
};
